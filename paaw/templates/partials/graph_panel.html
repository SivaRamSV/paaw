async init() { try { await this.loadData(); this.setupGraph(); this.hideLoading(); } catch (e) { console.error('Graph init failed:', e); this.showError(); } } async loadData() { const response = await fetch('/api/graph'); const data = await response.json(); this.nodes = data.nodes || []; this.links = data.edges || []; } setupGraph() { const width = this.container.clientWidth; const height = this.container.clientHeight || 300; // Clear this.svg.selectAll('*').remove(); // Setup zoom const g = this.svg.append('g'); this.svg.call(d3.zoom() .scaleExtent([0.2, 4]) .on('zoom', (e) => g.attr('transform', e.transform))); // Arrow marker this.svg.append('defs').append('marker') .attr('id', 'arrow') .attr('viewBox', '0 -5 10 10') .attr('refX', 20) .attr('refY', 0) .attr('markerWidth', 6) .attr('markerHeight', 6) .attr('orient', 'auto') .append('path') .attr('d', 'M0,-5L10,0L0,5') .attr('fill', '#1d3557'); // Filter nodes const visibleNodes = this.getVisibleNodes(); const visibleIds = new Set(visibleNodes.map(n => n.id)); const visibleLinks = this.links.filter(l => visibleIds.has(l.source) && visibleIds.has(l.target) ); // Simulation this.simulation = d3.forceSimulation(visibleNodes) .force('link', d3.forceLink(visibleLinks).id(d => d.id).distance(60)) .force('charge', d3.forceManyBody().strength(-150)) .force('center', d3.forceCenter(width / 2, height / 2)) .force('collision', d3.forceCollide().radius(25)); // Links const link = g.append('g') .selectAll('line') .data(visibleLinks) .join('line') .attr('stroke', '#1d3557') .attr('stroke-width', 1) .attr('marker-end', 'url(#arrow)'); // Nodes const node = g.append('g') .selectAll('circle') .data(visibleNodes) .join('circle') .attr('r', d => d.type === 'User' ? 12 : d.type === 'Memory' ? 5 : 8) .attr('fill', d => this.colors[d.type] || this.colors.default) .attr('stroke', '#0a192f') .attr('stroke-width', 2) .style('cursor', 'pointer') .call(this.drag()) .on('click', (e, d) => this.showNodeDetails(d)) .on('mouseover', function() { d3.select(this).attr('r', d => (d.type === 'Memory' ? 7 : 12)); }) .on('mouseout', function(e, d) { d3.select(this).attr('r', d.type === 'User' ? 12 : d.type === 'Memory' ? 5 : 8); }); // Labels (only for non-Memory nodes to reduce clutter) const label = g.append('g') .selectAll('text') .data(visibleNodes.filter(n => n.type !== 'Memory' && n.type !== 'Trail')) .join('text') .attr('font-size', '9px') .attr('fill', '#ccd6f6') .attr('text-anchor', 'middle') .attr('dy', 20) .text(d => d.label?.substring(0, 15) || d.id.substring(0, 15)); // Tick this.simulation.on('tick', () => { link .attr('x1', d => d.source.x) .attr('y1', d => d.source.y) .attr('x2', d => d.target.x) .attr('y2', d => d.target.y); node .attr('cx', d => d.x) .attr('cy', d => d.y); label .attr('x', d => d.x) .attr('y', d => d.y); }); } getVisibleNodes() { let nodes = this.nodes; // Filter by type if (this.activeFilter !== 'all') { nodes = nodes.filter(n => n.type === this.activeFilter); } // Filter by search if (this.searchTerm) { const term = this.searchTerm.toLowerCase(); nodes = nodes.filter(n => n.label?.toLowerCase().includes(term) || n.id?.toLowerCase().includes(term) || n.context?.toLowerCase().includes(term) ); } return nodes; } drag() { return d3.drag() .on('start', (e, d) => { if (!e.active) this.simulation.alphaTarget(0.3).restart(); d.fx = d.x; d.fy = d.y; }) .on('drag', (e, d) => { d.fx = e.x; d.fy = e.y; }) .on('end', (e, d) => { if (!e.active) this.simulation.alphaTarget(0); d.fx = null; d.fy = null; }); } filter(type) { this.activeFilter = type; // Update button styles document.querySelectorAll('.filter-btn').forEach(btn => { if (btn.dataset.filter === type) { btn.className = 'filter-btn active px-2 py-1 rounded text-xs bg-teal-400 text-navy-900'; } else { btn.className = 'filter-btn px-2 py-1 rounded text-xs bg-navy-700 text-slate-300 hover:bg-teal-400 hover:text-navy-900'; } }); this.setupGraph(); } search(term) { this.searchTerm = term; this.setupGraph(); } showNodeDetails(node) { const detailsDiv = document.getElementById('node-details'); const typeColor = this.colors[node.type] || this.colors.default; detailsDiv.innerHTML = `
${node.label || node.id}
${node.type}
${node.context ? `
${node.context.substring(0, 200)}${node.context.length > 200 ? '...' : ''}
` : ''} ${node.key_facts?.length ? `
Key Facts:
    ${node.key_facts.slice(0, 3).map(f => `
  • ${f.substring(0, 50)}...
    • `).join('')}
` : ''}
`; } highlightNodes(nodeIds) { // Highlight specific nodes (called from chat) this.svg.selectAll('circle') .transition() .duration(300) .attr('r', d => nodeIds.includes(d.id) ? 16 : (d.type === 'Memory' ? 5 : 8)) .attr('stroke', d => nodeIds.includes(d.id) ? '#fff' : '#0a192f') .attr('stroke-width', d => nodeIds.includes(d.id) ? 3 : 2); setTimeout(() => { this.svg.selectAll('circle') .transition() .duration(300) .attr('r', d => d.type === 'User' ? 12 : d.type === 'Memory' ? 5 : 8) .attr('stroke', '#0a192f') .attr('stroke-width', 2); }, 3000); } async refresh() { // Reload data and redraw graph (called after chat messages) try { await this.loadData(); this.setupGraph(); console.log('Graph refreshed:', this.nodes.length, 'nodes'); } catch (e) { console.error('Graph refresh failed:', e); } } hideLoading() { document.getElementById('graph-loading').style.display = 'none'; } showError() { document.getElementById('graph-loading').style.display = 'none'; document.getElementById('graph-error').classList.remove('hidden'); } } // Initialize when DOM is ready document.addEventListener('DOMContentLoaded', () => { window.graphManager = new GraphManager(); });
